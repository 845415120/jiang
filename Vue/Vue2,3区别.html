子传父 emit ​

首先，在子组件中定义一个方法，该方法触发一个自定义事件并传递值

在父组件中使用子组件并监听该自定义事件

vue2

vue
App.vue

import AnswerButton from "./AnswerButton.vue";
export default {
  components: {
    AnswerButton,
  },

    return {
      canCome: true,
    };
  },
  methods: {
    onAnswerNo() {
      this.canCome = false;

      this.canCome = true;

  },

      {{ canCome ? "😀" : "😥" }}
    </p>
  </div>
</template>

组合选项 provide​ ​

用于提供可以被后代组件注入的值。 provide 和inject 通常成对一起使用，使一个祖先组件作为其后代组件的依赖注入方，无论这个组件的层级有多深都可以注入成功，只要他们处于同一条组件链上。

vue3

vue
<script setup>
import { ref, provide } from "vue";
import UserProfile from "./UserProfile.vue";

const user = ref({
  id: 1,
  username: "unicorn42",
  email: "unicorn42@example.com",
});

function updateUsername(username) {
  user.value.username = username;
}

provide("user", { user, updateUsername });
</script>

<template>
  <h1>Welcome back, {{ user.username }}</h1>
  <UserProfile />
</template>

//UserProfile.vue

<script setup>
import { inject } from "vue";
const { user, updateUsername } = inject("user");
</script>
<template>
  <div>
    <h2>My Profile</h2>
    <p>Username: {{ user.username }}</p>
    <p>Email: {{ user.email }}</p>
    <button @click="() => updateUsername('Jane')">
      Update username to Jane
    </button>
  </div>
</template>

  </p>

  </p>

    >
      <img
        :src="user.picture.thumbnail"

        {{ user.name.last }}
      </p>
    </li>
  </ul>

vue
import { ref } from "vue";

export default function useFetchUsers() {
  const data = ref();
  const error = ref();
  const isLoading = ref(false);

  async function fetchData() {
    isLoading.value = true;
    try {
      const response = await fetch("https://randomuser.me/api/?results=3");
      const { results: users } = await response.json();
      data.value = users;
      error.value = undefined;
    } catch (err) {
      data.value = undefined;
      error.value = err;
    }
    isLoading.value = false;
  }
  fetchData();

  return { isLoading, error, data };
}
